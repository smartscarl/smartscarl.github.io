របៀបដែល WPM និងភាពត្រឹមត្រូវត្រូវបានគណនា

អនុសញ្ញា​ស្តង់ដារ​រាប់​ពាក្យ​មួយ​ជា​ការ​ចុច​គ្រាប់ចុច​ប្រាំ រួម​ទាំង​ចន្លោះ។ សរុប WPM = (តួអក្សរដែលបានវាយ / 5) / នាទីបានកន្លងផុតទៅ។ សុទ្ធ WPM ដកកំហុសដែលមិនបានកែតម្រូវ៖ Net WPM = Gross WPM - (កំហុស/នាទី)។ ភាពត្រឹមត្រូវ = តួអក្សរត្រឹមត្រូវ / សរុបវាយ x 100% ។ សម្រាប់ភាសាចិន ល្បឿនត្រូវបានរាយការណ៍ជាញឹកញាប់ជាតួអក្សរក្នុងមួយនាទី ពីព្រោះតួអក្សរចិនមួយផ្ទុកព័ត៌មានច្រើនជាងការចុចគ្រាប់ចុចឡាតាំងតែមួយ។
